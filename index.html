    // CAROUSEL

    (function initCarousel() {

      const track = document.getElementById('cfTrack');

      if (!track) return;

      const items = Array.from(track.querySelectorAll('.carousel-cf-item'));

      const total = items.length;

      let offset = 0, gap = 0;


      function updateGap() {

        const w = items[0]?.getBoundingClientRect().width || window.innerWidth * 0.2;

        gap = window.innerWidth < 700 ? w * 1.35 : w * 1.1;

      }

      function render() {

        items.forEach((item, i) => {

          const raw = i - offset;

          const pos = ((raw % total) + total + total / 2) % total - total / 2;

          const tx = pos * gap, ty = pos * pos * 18, tz = -Math.abs(pos) * 180;

          const rotateY = Math.max(-65, Math.min(65, pos * -38));

          const scale = Math.max(0.6, 1 - Math.abs(pos) * 0.08);

          

          item.style.transform = `translate(-50%, -50%) translateX(${tx}px) translateY(${ty}px) translateZ(${tz}px) rotateY(${rotateY}deg) scale(${scale})`;

          item.style.opacity = Math.abs(pos) > 4 ? 0 : Math.max(0, 1 - Math.abs(pos) * 0.25);

          item.style.zIndex = Math.round(scale * 100);

        });

      }

      function loop(ts) {

        if (!loop.last) loop.last = ts;

        offset += 0.00022 * (ts - loop.last);

        loop.last = ts;

        render(); requestAnimationFrame(loop);

      }

      window.addEventListener('resize', updateGap, { passive: true });

      window.addEventListener('load', updateGap);

      updateGap(); render(); requestAnimationFrame(loop);

    })();
